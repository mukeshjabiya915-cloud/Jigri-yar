`; // --- BACKEND LOGIC --- app.get('/', (req, res) => { res.send(htmlContent); }); io.on('connection', (socket) => { console.log('User Connected: ' + socket.id); socket.on('chat_msg', (data) => { io.emit('chat_msg', data); }); socket.on('disconnect', () => { console.log('User Disconnected'); }); }); const PORT = 3000; server.listen(PORT, () => { console.log("सर्वर यहाँ चल रहा है: http://localhost:3000"); });
