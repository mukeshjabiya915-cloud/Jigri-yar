const express = require('express'); const http = require('http'); const { Server } = require('socket.io'); const app = express(); const server = http.createServer(app); const io = new Server(server); // --- FRONTEND SECTION (HTML/CSS/JS) --- const htmlContent = ` Live Chat App
Live Chat - Ravi Boss
`; // --- BACKEND LOGIC --- app.get('/', (req, res) => { res.send(htmlContent); }); io.on('connection', (socket) => { console.log('User Connected: ' + socket.id); socket.on('chat_msg', (data) => { io.emit('chat_msg', data); // सबको मैसेज भेजें }); socket.on('disconnect', () => { console.log('User Disconnected'); }); }); const PORT = 3000; server.listen(PORT, () => { console.log("सर्वर यहाँ चल रहा है: http://localhost:3000"); });
