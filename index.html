const express = require('express'); const app = express(); const http = require('http').Server(app); const io = require('socket.io')(http); const path = require('path'); app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'index.html')); }); let users = {}; // सभी ऑनलाइन यूजर्स let waitingUser = null; // इंतज़ार करने वाला यूजर io.on('connection', (socket) => { console.log('एक नया यूजर जुड़ गया: ' + socket.id); socket.on('join', (data) => { users[socket.id] = { id: socket.id, name: data.name, gender: data.gender }; // मैचमेकिंग लॉजिक: अगर कोई पहले से इंतज़ार कर रहा है तो उसे जोड़ें if (waitingUser && waitingUser !== socket.id) { const partnerId = waitingUser; waitingUser = null; // दोनों को एक दूसरे की ID भेजें ताकि कॉल कनेक्ट हो सके io.to(socket.id).emit('matched', { partnerId: partnerId, partnerName: users[partnerId].name }); io.to(partnerId).emit('matched', { partnerId: socket.id, partnerName: data.name }); } else { waitingUser = socket.id; socket.emit('waiting', 'पार्टनर की तलाश जारी है...'); } }); socket.on('disconnect', () => { delete users[socket.id]; if (waitingUser === socket.id) waitingUser = null; console.log('यूजर चला गया'); }); }); http.listen(3000, () => { console.log('सर्वर चालू है: http://localhost:3000'); });
