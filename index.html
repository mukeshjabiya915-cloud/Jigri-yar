const express = require('express'); const http = require('http'); const { Server } = require('socket.io'); const app = express(); const server = http.createServer(app); const io = new Server(server); // जब कोई यूजर कनेक्ट होता है io.on('connection', (socket) => { console.log('एक यूजर जुड़ गया है:', socket.id); // मैसेज प्राप्त करना और सबको भेजना socket.on('send_message', (data) => { io.emit('receive_message', data); }); socket.on('disconnect', () => { console.log('यूजर चला गया'); }); }); server.listen(3001, () => { console.log("सर्वर पोर्ट 3001 पर चल रहा है..."); });
